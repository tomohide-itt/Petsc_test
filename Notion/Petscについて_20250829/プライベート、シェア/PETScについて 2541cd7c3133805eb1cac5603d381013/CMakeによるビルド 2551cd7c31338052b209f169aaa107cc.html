CMakeによるビルド

連立方程式を解く最小コード で作成した poisson.c をCMakeでビルドする

CMakeLists.txt
cmake_minimum_required(VERSION 3.18)
set( PRG_NAME "poisson" )
project( ${PRG_NAME} C)
list(APPEND CMAKE_PREFIX_PATH "$ENV{PETSC_DIR}/$ENV{PETSC_ARCH}")
find_package(PETSc REQUIRED MODULE)
add_executable(${PRG_NAME} ${PRG_NAME}.c)
target_link_libraries(${PRG_NAME} PRIVATE PETSc::petsc)

上記では，うまくビルドできなかったので，以下のように修正

cmake_minimum_required(VERSION 3.18)
set( PRG_NAME "poisson" )
project( ${PRG_NAME} C)
find_package(PkgConfig REQUIRED)
pkg_check_modules(PETSC REQUIRED IMPORTED_TARGET petsc)
add_executable(${PRG_NAME} ${PRG_NAME}.c)
target_link_libraries(${PRG_NAME} PRIVATE PkgConfig::PETSC)

これであればビルドできた．原因は不明だが，これで良いか．

ビルドと実行
mkdir build && cd build
cmake ..
make
mpiexec -n 2 ./poisson -ksp_monitor -ksp_converged_reason
