cppファイルにしてみる

今までは， ソースファイルを***.c としていたが， ***.cpp としてビルド＆実行ができるか試してみる．

CMakeLists.txtの変更

gmsh1.c の内容はそのままで，拡張子だけ変更して gmsh2.cpp とし，CMakeLists.txtを以下のように変更する．

cmake_minimum_required(VERSION 3.18)
set( PRG_NAME "gmsh2" )
project( ${PRG_NAME} CXX)
find_package(MPI REQUIRED)
find_package(PkgConfig REQUIRED)
pkg_check_modules(PETSC REQUIRED IMPORTED_TARGET petsc)
add_executable(${PRG_NAME} ${PRG_NAME}.cpp)
target_link_libraries(${PRG_NAME} PRIVATE MPI::MPI_CXX PkgConfig::PETSC)

変更点は以下の通り．

  • 3行目：CXX を指定
  • 7行目：ソースファイルの拡張子を.cppに変更
  • 8行目：リンクするライブラリを MPI::MPI_C から MPI::MPI_CXX に変更
ビルド＆実行
mkdir build && cd build
cmake ..
make

mpiexec -n 4 ./gmsh2 \
-dm_plex_filename ../gmsh_input/test2D.msh \
-dm_plex_interpolate 1 \
-petscpartitioner_type parmetis

実行結果は， gmsh1.c と同じ

mshファイル名をコード内で指定する

mshファイル名は -dm_plex_filename ***.msh のように実行時にオプションで指定していたが，コード内で指定するように変更する

  PetscCall(DMCreate(PETSC_COMM_WORLD, &dm));
  PetscCall(DMSetType(dm, DMPLEX));
  PetscCall(DMSetFromOptions(dm));      // <- ここで -dm_plex_* オプションが効く
  PetscCall(DMViewFromOptions(dm, NULL, "-dm_view"));

の部分を以下のように変更

std::string fmsh_name = "../gmsh_input/test2D.msh";
 PetscCall( DMPlexCreateGmshFromFile( PETSC_COMM_WORLD, fmsh_name.c_str(), PETSC_TRUE, &dm ) );

ここで， DIMPlexCreateGmshFromFile() の第３引数は，interpolateフラグであり，

  • PETSC_TRUE なら頂点・稜線・面・セルを作成して接続も構築する

    実行時オプションで -dm_plex_interpolate 1 と指定することに対応

  • PETSC_FALSE ならセルと頂点のみを作成

    -dm_plex_interpolate 0 と指定することに対応

実行時は，今まで指定していた -dm_plex_filename と -dm_plex_interpolate が不要になるため，以下のように実行する．

mpiexec -n 4 ./gmsh2 -petscpartitioner_type parmetis
領域分割方法をコード内で指定する
// 並列分割（コードから明示的に行う）
  if (size > 1) {
    PetscPartitioner part;
    PetscCall(DMPlexGetPartitioner(dm, &part));
    // 既定は "simple"。ParMETIS/PT-Scotchを使うなら実行時オプションで変更推奨。
    // 例: -petscpartitioner_type parmetis
    PetscCall(PetscPartitionerSetFromOptions(part)); 
    PetscCall(DMPlexDistribute(dm, 0, NULL, &dmDist));
    if (dmDist) { PetscCall(DMDestroy(&dm)); dm = dmDist; }
  }

の部分を以下のように変更

// 並列分割（コードから明示的に行う）
  if (size > 1) {
    PetscPartitioner part;
    PetscCall(DMPlexGetPartitioner(dm, &part));
    PetscCall(PetscPartitionerSetType(part, PETSCPARTITIONERPARMETIS));
    PetscCall(DMPlexDistribute(dm, 0, NULL, &dmDist));
    if (dmDist) { PetscCall(DMDestroy(&dm)); dm = dmDist; }
  }

PetscCall(PetscPartitionerSetType(part, PETSCPARTITIONERPARMETIS)); で，PartitionerにParMETISを指定している．

実行時オプション -petscpartitioner_type parmetis は不要となり，以下のように実行する

mpiexec -n 4 ./gmsh2
